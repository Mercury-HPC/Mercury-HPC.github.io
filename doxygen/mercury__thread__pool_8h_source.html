74 static HG_UTIL_INLINE int
75 hg_thread_pool_post(hg_thread_pool_t *pool, struct hg_thread_work *work)
76 {
77  int ret = HG_UTIL_SUCCESS;
78 
79  if (!pool) {
80  HG_UTIL_LOG_ERROR("Thread pool not initialized");
81  ret = HG_UTIL_FAIL;
82  goto done;
83  }
84 
85  if (!work) {
86  HG_UTIL_LOG_ERROR("Thread work cannot be NULL");
87  ret = HG_UTIL_FAIL;
88  goto done;
89  }
90 
91  if (!work->func) {
92  HG_UTIL_LOG_ERROR("Function pointer cannot be NULL");
93  ret = HG_UTIL_FAIL;
94  goto done;
95  }
96 
97  hg_thread_mutex_lock(&pool->mutex);
98 
99  /* Are we shutting down ? */
100  if (pool->shutdown) {
101  HG_UTIL_LOG_ERROR("Pool is shutting down");
102  ret = HG_UTIL_FAIL;
103  goto unlock;
104  }
105 
106  /* Add task to task queue */
107  HG_QUEUE_PUSH_TAIL(&pool->queue, work, entry);
108 
109  /* Wake up sleeping worker */
110  if (pool->sleeping_worker_count) {
111  if (hg_thread_cond_signal(&pool->cond) != HG_UTIL_SUCCESS) {
112  HG_UTIL_LOG_ERROR("Cannot signal pool condition");
113  ret = HG_UTIL_FAIL;
114  }
115  }
116 
117 unlock:
118  hg_thread_mutex_unlock(&pool->mutex);
119 
120 done:
121  return ret;
122 }
