43 /* Generate structure */
44 #define HG_GEN_STRUCT(struct_type_name, fields) \
45 typedef struct \

47  BOOST_PP_SEQ_FOR_EACH(HG_GEN_STRUCT_FIELD, , fields) \

51 /* Generate proc for struct field */
52 #define HG_GEN_PROC(r, struct_name, field) \
53  ret = BOOST_PP_CAT(hg_proc_, HG_GEN_GET_TYPE(field) \
54  (proc, &struct_name->HG_GEN_GET_NAME(field))); \
55  if (ret != HG_SUCCESS) { \
56  HG_LOG_ERROR("Proc error"); \
57  return ret; \
58  }

60 /* Generate proc for struct */

62 static HG_INLINE hg_return_t \
63  BOOST_PP_CAT(hg_proc_, struct_type_name) \
64  (hg_proc_t proc, void *data) \

66  hg_return_t ret = HG_SUCCESS; \
67  struct_type_name *struct_data = (struct_type_name *) data; \
68  \
69  BOOST_PP_SEQ_FOR_EACH(HG_GEN_PROC, struct_data, fields) \
