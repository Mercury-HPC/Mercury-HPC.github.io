178 static HG_UTIL_INLINE int
179 hg_thread_cond_timedwait(hg_thread_cond_t *cond, hg_thread_mutex_t *mutex,
180  unsigned int timeout)
181 {
182  int ret = HG_UTIL_SUCCESS;
183 
184 #ifdef _WIN32
185  if (!SleepConditionVariableCS(cond, mutex, timeout)) ret = HG_UTIL_FAIL;
186 #else
187  int pret;
188 # if defined(HG_UTIL_HAS_PTHREAD_CONDATTR_SETCLOCK)
189  hg_time_t now;
190 # elif defined(HG_UTIL_HAS_SYSTIME_H)
191  struct timeval now;
192 # endif
193  struct timespec abs_timeout;
194  long int abs_timeout_us;
195  ldiv_t ld;
196 
197  /* Need to convert timeout (ms) to absolute time */
198 # if defined(HG_UTIL_HAS_PTHREAD_CONDATTR_SETCLOCK)
199  if (hg_time_get_current(&now) != HG_UTIL_SUCCESS)
200  return HG_UTIL_FAIL;
201 # elif defined(HG_UTIL_HAS_SYSTIME_H)
202  if (gettimeofday(&now, NULL) != 0)
203  return HG_UTIL_FAIL;
204 # endif
205  abs_timeout_us = now.tv_usec + timeout * 1000L;
206  /* Get sec / nsec */
207  ld = ldiv(abs_timeout_us, 1000000L);
208  abs_timeout.tv_sec = now.tv_sec + ld.quot;
209  abs_timeout.tv_nsec = ld.rem * 1000L;
210 
211  pret = pthread_cond_timedwait(cond, mutex, &abs_timeout);
212  if (pret)
213  ret = HG_UTIL_FAIL;
214 #endif
215 
216  return ret;
217 }
