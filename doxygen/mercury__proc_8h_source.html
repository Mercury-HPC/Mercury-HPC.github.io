307 static HG_INLINE void *
308 hg_proc_buf_memcpy(void *buf, void *data, hg_size_t data_size, hg_proc_op_t op)
309 {
310  const void *src = NULL;
311  void *dest = NULL;
312 
313  if ((op != HG_ENCODE) && (op != HG_DECODE)) return NULL;
314  src = (op == HG_ENCODE) ? (const void *) data : (const void *) buf;
315  dest = (op == HG_ENCODE) ? buf : data;
316  memcpy(dest, src, data_size);
317 
318  return ((char *) buf + data_size);
319 }

536 HG_PROC_INLINE hg_return_t
537 hg_proc_raw(hg_proc_t proc, void *buf, hg_size_t buf_size)
538 {
539  hg_uint8_t *buf_ptr;
540  hg_uint8_t *buf_ptr_lim = (hg_uint8_t*) buf + buf_size;
541  hg_return_t ret = HG_SUCCESS;
542 
543  for (buf_ptr = (hg_uint8_t*) buf; buf_ptr < buf_ptr_lim; buf_ptr++) {
544  ret = hg_proc_uint8_t(proc, buf_ptr);
545  if (ret != HG_SUCCESS) {
546  HG_LOG_ERROR("Proc error");
547  break;
548  }
549  }
550 
551  return ret;
552 }

562 HG_PROC_INLINE hg_return_t
563 hg_proc_hg_bulk_t(hg_proc_t proc, hg_bulk_t *handle)
564 {
565  hg_return_t ret = HG_SUCCESS;
566  void *buf = NULL;
567  hg_uint64_t buf_size = 0;
568 
569  switch (hg_proc_get_op(proc)) {
570  case HG_ENCODE:
571  if (*handle != HG_BULK_NULL) {
572  hg_bool_t request_eager = HG_FALSE;
573 #ifdef HG_HAS_EAGER_BULK
574  request_eager = (hg_proc_get_size_left(proc)
575  > HG_Bulk_get_serialize_size(*handle, HG_TRUE))
576  ? HG_TRUE : HG_FALSE;
577 #endif
578  buf_size = HG_Bulk_get_serialize_size(*handle, request_eager);
579  buf = malloc(buf_size);
580  ret = HG_Bulk_serialize(buf, buf_size, request_eager, *handle);
581  if (ret != HG_SUCCESS) {
582  HG_LOG_ERROR("Could not serialize bulk handle");
583  return ret;
584  }
585  } else {
586  /* If HG_BULK_NULL set 0 to buf_size */
587  buf_size = 0;
588  }
589  /* Encode size */
590  ret = hg_proc_uint64_t(proc, &buf_size);
591  if (ret != HG_SUCCESS) {
592  HG_LOG_ERROR("Proc error");
593  return ret;
594  }
595  if (buf_size) {
596  /* Encode serialized buffer */
597  ret = hg_proc_raw(proc, buf, buf_size);
598  if (ret != HG_SUCCESS) {
599  HG_LOG_ERROR("Proc error");
600  return ret;
601  }
602  free(buf);
603  buf = NULL;
604  }
605  break;
606  case HG_DECODE:
607  /* Decode size */
608  ret = hg_proc_uint64_t(proc, &buf_size);
609  if (ret != HG_SUCCESS) {
610  HG_LOG_ERROR("Proc error");
611  return ret;
612  }
613  if (buf_size) {
614  hg_class_t *hg_class = hg_proc_get_class(proc);
615 
616  buf = malloc(buf_size);
617  /* Decode serialized buffer */
618  ret = hg_proc_raw(proc, buf, buf_size);
619  if (ret != HG_SUCCESS) {
620  HG_LOG_ERROR("Proc error");
621  return ret;
622  }
623  ret = HG_Bulk_deserialize(hg_class, handle, buf, buf_size);
624  if (ret != HG_SUCCESS) {
625  HG_LOG_ERROR("Could not deserialize bulk handle");
626  return ret;
627  }
628  free(buf);
629  buf = NULL;
630  } else {
631  /* If buf_size is 0, define handle to HG_BULK_NULL */
632  *handle = HG_BULK_NULL;
633  }
634  break;
635  case HG_FREE:
636  if (*handle != HG_BULK_NULL) {
637  ret = HG_Bulk_free(*handle);
638  if (ret != HG_SUCCESS) {
639  HG_LOG_ERROR("Could not free bulk handle");
640  return ret;
641  }
642  *handle = HG_BULK_NULL;
643  } else {
644  /* If *handle is HG_BULK_NULL, just return success */
645  ret = HG_SUCCESS;
646  }
647  break;
648  default:
649  break;
650  }
651  return ret;
652 }
