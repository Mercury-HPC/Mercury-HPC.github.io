82 /* Check whether size exceeds current proc size left */
83 #ifdef HG_HAS_XDR
84 # define HG_PROC_CHECK_SIZE(proc, size, label, ret) \
85  do { \
86  if (unlikely(((struct hg_proc *) proc)->current_buf->size_left < \
87  size)) { \
88  ret = HG_OVERFLOW; \
89  goto label; \
90  } \
91  } while (0)
92 #else
93 # define HG_PROC_CHECK_SIZE(proc, size, label, ret) \
94  do { \
95  if (unlikely(((struct hg_proc *) proc)->current_buf->size_left < \
96  size)) { \
97  ret = hg_proc_set_size(proc, hg_proc_get_size(proc) + size); \
98  if (ret != HG_SUCCESS) \
99  goto label; \
100  } \
101  } while (0)
102 #endif

104 /* Encode type */
105 #define HG_PROC_TYPE_ENCODE(proc, data, size) \
106  memcpy(((struct hg_proc *) proc)->current_buf->buf_ptr, data, size)

108 /* Decode type */

112 /* Update proc pointers */

114  do { \

116  (char *) ((struct hg_proc *) proc)->current_buf->buf_ptr + size; \
117  ((struct hg_proc *) proc)->current_buf->size_left -= size; \
118  } while (0)

128 /* Base proc function */
129 #ifdef HG_HAS_XDR
130 # define HG_PROC_TYPE(proc, type, data, label, ret) \
131  do { \
132  HG_PROC_CHECK_SIZE(proc, sizeof(type), label, ret); \
133  \
134  if (xdr_##type(hg_proc_get_xdr_ptr(proc), data) == 0) { \
135  ret = HG_PROTOCOL_ERROR; \
136  goto label; \
137  } \
138  \
139  HG_PROC_UPDATE(proc, sizeof(type)); \
140  HG_PROC_CHECKSUM_UPDATE(proc, data, sizeof(type)); \
141  } while (0)
142 #else
143 # define HG_PROC_TYPE(proc, type, data, label, ret) \
144  do { \
145  /* Do nothing in HG_FREE for basic types */ \
146  if (hg_proc_get_op(proc) == HG_FREE) \
147  goto label; \
148  \
149  /* If not enough space allocate extra space if encoding or just */ \
150  /* get extra buffer if decoding */ \
151  HG_PROC_CHECK_SIZE(proc, sizeof(type), label, ret); \
152  \
153  /* Encode, decode type */ \
154  if (hg_proc_get_op(proc) == HG_ENCODE) \
155  HG_PROC_TYPE_ENCODE(proc, data, sizeof(type)); \
156  else \
157  HG_PROC_TYPE_DECODE(proc, data, sizeof(type)); \
158  \
159  /* Update proc pointers etc */ \
160  HG_PROC_UPDATE(proc, sizeof(type)); \
161  HG_PROC_CHECKSUM_UPDATE(proc, data, sizeof(type)); \
162  } while (0)
163 #endif
164

167 # define HG_PROC_BYTES(proc, data, size, label, ret) \
168  do { \

170  \
171  if (xdr_bytes(hg_proc_get_xdr_ptr(proc), (char **) &data, \
172  (u_int *) &size, UINT_MAX) == 0) { \
173  ret = HG_PROTOCOL_ERROR; \
174  goto label; \
175  } \
