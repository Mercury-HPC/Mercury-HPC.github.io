563 HG_PROC_INLINE hg_return_t
564 hg_proc_hg_bulk_t(hg_proc_t proc, void *handle)
565 {
566  hg_return_t ret = HG_SUCCESS;
567  void *buf = NULL;
568  hg_bulk_t *bulk = (hg_bulk_t*)handle;
569  hg_uint64_t buf_size = 0;
570 
571  switch (hg_proc_get_op(proc)) {
572  case HG_ENCODE: {
573  hg_bool_t request_eager = HG_FALSE;
574 
575  if (*bulk == HG_BULK_NULL) {
576  /* If HG_BULK_NULL set 0 to buf_size */
577  buf_size = 0;
578  } else {
579 #ifdef HG_HAS_EAGER_BULK
580  request_eager = (hg_proc_get_size_left(proc)
581  > HG_Bulk_get_serialize_size(*bulk, HG_TRUE))
582  ? HG_TRUE : HG_FALSE;
583 #endif
584  buf_size = HG_Bulk_get_serialize_size(*bulk, request_eager);
585  }
586  /* Encode size */
587  ret = hg_proc_uint64_t(proc, &buf_size);
588  if (ret != HG_SUCCESS) {
589  HG_LOG_ERROR("Proc error");
590  return ret;
591  }
592  if (buf_size) {
593  buf = hg_proc_save_ptr(proc, buf_size);
594  ret = HG_Bulk_serialize(buf, buf_size, request_eager, *bulk);
595  if (ret != HG_SUCCESS) {
596  HG_LOG_ERROR("Could not serialize bulk handle");
597  return ret;
598  }
599  hg_proc_restore_ptr(proc, buf, buf_size);
600  }
601  }
602  break;
603  case HG_DECODE:
604  /* Decode size */
605  ret = hg_proc_uint64_t(proc, &buf_size);
606  if (ret != HG_SUCCESS) {
607  HG_LOG_ERROR("Proc error");
608  return ret;
609  }
610  if (buf_size) {
611  hg_class_t *hg_class = hg_proc_get_class(proc);
612 
613  buf = hg_proc_save_ptr(proc, buf_size);
614  ret = HG_Bulk_deserialize(hg_class, bulk, buf, buf_size);
615  if (ret != HG_SUCCESS) {
616  HG_LOG_ERROR("Could not deserialize bulk handle");
617  return ret;
618  }
619  hg_proc_restore_ptr(proc, buf, buf_size);
620  } else {
621  /* If buf_size is 0, define handle to HG_BULK_NULL */
622  *bulk = HG_BULK_NULL;
623  }
624  break;
625  case HG_FREE:
626  if (*bulk != HG_BULK_NULL) {
627  ret = HG_Bulk_free(*bulk);
628  if (ret != HG_SUCCESS) {
629  HG_LOG_ERROR("Could not free bulk handle");
630  return ret;
631  }
632  *bulk = HG_BULK_NULL;
633  } else {
634  /* If *bulk is HG_BULK_NULL, just return success */
635  ret = HG_SUCCESS;
636  }
637  break;
638  default:
639  break;
640  }
641  return ret;
642 }

655 HG_PROC_INLINE void *
656 hg_proc_buf_memcpy(void *buf, void *data, hg_size_t data_size, hg_proc_op_t op)
657 {
658  const void *src = NULL;
659  void *dest = NULL;
660 
661  if ((op != HG_ENCODE) && (op != HG_DECODE)) return NULL;
662  src = (op == HG_ENCODE) ? (const void *) data : (const void *) buf;
663  dest = (op == HG_ENCODE) ? buf : data;
664  memcpy(dest, src, data_size);
665 
666  return ((char *) buf + data_size);
667 }
