219 /* Generate RPC stub */
220 #define MERCURY_GEN_RPC_STUB(gen_func_name, func_name, with_ret, \
221  ret_type_name, ret_fail, with_input, in_struct_type_name, in_params, \
222  with_output, out_struct_type_name, out_params, with_bulk, bulk_read) \
223  BOOST_PP_IF(with_ret, ret_type_name, void) \
224  gen_func_name HG_GEN_DECL_FUNC_PARAMS(with_input, in_params, \
225  BOOST_PP_IF(with_bulk, HG_BULK_EXTRA_IN_PARAM, BOOST_PP_EMPTY()), \
226  with_output, out_params, ) \
227  { \
228  BOOST_PP_IF(with_input, in_struct_type_name in_struct; \
229  , BOOST_PP_EMPTY()) \
230  BOOST_PP_IF(BOOST_PP_OR(with_output, with_ret), \
231  out_struct_type_name out_struct; \
232  , BOOST_PP_EMPTY()) \
233  BOOST_PP_IF(with_ret, ret_type_name ret;, BOOST_PP_EMPTY()) \
234  hg_id_t id; \
235  hg_handle_t handle; \
236  BOOST_PP_IF( \
237  with_bulk, HG_GEN_DECL_PARAMS(HG_BULK_PARAM), BOOST_PP_EMPTY()) \
238  hg_bool_t func_registered; \
239  hg_return_t hg_ret; \
240  \
241  /* Init stack if not initialized */ \
242  HG_Hl_init(NULL, 0); \
243  \
244  /* Check whether call has already been registered or not */ \
245  HG_Registered_rpc(HG_CLASS_DEFAULT, BOOST_PP_STRINGIZE(func_name), \
246  &func_registered, &id); \
247  if (!func_registered) { \
248  id = MERCURY_REGISTER(HG_CLASS_DEFAULT, \
249  BOOST_PP_STRINGIZE(func_name), \
250  BOOST_PP_IF(with_input, in_struct_type_name, void), \
251  BOOST_PP_IF(BOOST_PP_OR(with_output, with_ret), \
252  out_struct_type_name, void), \
253  NULL); \
254  } \
255  \
256  /* Create HG handle */ \
257  hg_ret = HG_Create(HG_CLASS_DEFAULT, HG_CONTEXT_DEFAULT, \
258  NA_ADDR_DEFAULT, id, &handle); \
259  if (hg_ret != HG_SUCCESS) { \
260  BOOST_PP_IF(with_ret, ret = ret_fail;, BOOST_PP_EMPTY()) \
261  goto done; \
262  } \
263  \
264  /* Create bulk handle */ \
265  BOOST_PP_IF(with_bulk, \
266  HG_BULK_REGISTER(handle, \
267  HG_GEN_GET_NAME(BOOST_PP_SEQ_HEAD(HG_BULK_PARAM)), with_ret, \
268  ret_fail, bulk_read), \
269  BOOST_PP_EMPTY()) \
270  \
271  /* Fill input structure */ \
272  BOOST_PP_IF(with_input, \
273  HG_SET_STRUCT_PARAMS( \
274  in_struct, in_params BOOST_PP_IF( \
275  with_bulk, HG_BULK_PARAM, BOOST_PP_EMPTY())), \
276  BOOST_PP_EMPTY()) \
277  \
278  /* Forward call to default target */ \
279  hg_ret = HG_Hl_forward_wait( \
280  handle, BOOST_PP_IF(with_input, &in_struct, NULL)); \
281  if (hg_ret != HG_SUCCESS) { \
282  BOOST_PP_IF(with_ret, ret = ret_fail;, BOOST_PP_EMPTY()) \
283  goto done; \
284  } \
285  \
286  /* Free bulk handle */ \
287  BOOST_PP_IF(with_bulk, \
288  HG_BULK_FREE(HG_GEN_GET_NAME(BOOST_PP_SEQ_HEAD(HG_BULK_PARAM)), \
289  with_ret, ret_fail), \
290  BOOST_PP_EMPTY()) \
291  \
292  /* Get output */ \
293  BOOST_PP_IF(BOOST_PP_OR(with_output, with_ret), \
294  HG_GET_OUTPUT(with_ret, ret_fail), BOOST_PP_EMPTY()) \
295  \
296  /* Get output parameters */ \
297  BOOST_PP_IF(with_ret, \
298  HG_GET_STRUCT_PARAMS(out_struct, ((ret_type)(ret))), \
299  BOOST_PP_EMPTY()) \
300  BOOST_PP_IF(with_output, \
301  HG_GET_OUT_STRUCT_PARAMS(out_struct, out_params), \
302  BOOST_PP_EMPTY()) \
303  \
304  /* Free output */ \
305  BOOST_PP_IF(BOOST_PP_OR(with_output, with_ret), \
306  HG_FREE_OUTPUT(with_ret, ret_fail), BOOST_PP_EMPTY()) \
307  \
308  /* Destroy handle */ \
309  hg_ret = HG_Destroy(handle); \
310  if (hg_ret != HG_SUCCESS) { \
311  BOOST_PP_IF(with_ret, ret = ret_fail;, BOOST_PP_EMPTY()) \
312  goto done; \
313  } \
314  \
315 done: \
316  \
317  return BOOST_PP_IF(with_ret, ret, BOOST_PP_EMPTY()); \
318  }

374  BOOST_PP_IF(with_ret, ret =, BOOST_PP_EMPTY()) \
375  func_name HG_GEN_FUNC_PARAMS(with_input, in_params, \
376  BOOST_PP_IF(with_bulk, HG_BULK_EXTRA_IN_PARAM, BOOST_PP_EMPTY()), \
377  with_output, out_params, ); \
378  \
379  BOOST_PP_IF(with_bulk, \
380  BOOST_PP_IF(bulk_read, BOOST_PP_EMPTY(), \
381  HG_BULK_TRANSFER(handle, \
382  HG_GEN_GET_NAME(BOOST_PP_SEQ_HEAD(HG_BULK_PARAM)), \
383  HG_GEN_GET_NAME(BOOST_PP_SEQ_HEAD(HG_BULK_LOCAL_PARAM)), \
