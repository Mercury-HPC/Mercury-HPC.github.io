26 hg_proc_hg_bulk_t(hg_proc_t proc, void *data)
27 {
28  hg_return_t ret = HG_SUCCESS;
29  void *buf = NULL;
30  hg_bulk_t *bulk_ptr = (hg_bulk_t *) data;
31  hg_uint64_t buf_size = 0;
32 
33  switch (hg_proc_get_op(proc)) {
34  case HG_ENCODE: {
35  hg_bool_t request_eager = HG_FALSE;
36  void *cached_ptr = NULL;
37 
38  /* If HG_BULK_NULL set 0 to buf_size */
39  if (*bulk_ptr == HG_BULK_NULL)
40  buf_size = 0;
41  else if ((cached_ptr = HG_Bulk_get_serialize_cached_ptr(*bulk_ptr))
42  != NULL)
43  buf_size = HG_Bulk_get_serialize_cached_size(*bulk_ptr);
44  else {
45 #ifdef HG_HAS_EAGER_BULK
46  hg_size_t serialize_size = HG_Bulk_get_serialize_size(*bulk_ptr,
47  HG_TRUE);
48  request_eager =
49  (hg_proc_get_size_left(proc) > serialize_size) ? HG_TRUE :
50  HG_FALSE;
51  if (request_eager)
52  buf_size = serialize_size;
53  else
54 #endif
55  buf_size = HG_Bulk_get_serialize_size(*bulk_ptr, HG_FALSE);
56  }
57  /* Encode size */
58  ret = hg_proc_uint64_t(proc, &buf_size);
59  if (ret != HG_SUCCESS) {
60  HG_LOG_ERROR("Proc error");
61  return ret;
62  }
63  if (!buf_size)
64  break;
65  if (cached_ptr)
66  hg_proc_raw(proc, cached_ptr, buf_size);
67  else {
68  buf = hg_proc_save_ptr(proc, buf_size);
69  ret = HG_Bulk_serialize(buf, buf_size, request_eager, *bulk_ptr);
70  if (ret != HG_SUCCESS) {
71  HG_LOG_ERROR("Could not serialize bulk handle");
72  return ret;
73  }
74  hg_proc_restore_ptr(proc, buf, buf_size);
75  }
76  break;
77  }
78  case HG_DECODE: {
79  hg_class_t *hg_class = hg_proc_get_class(proc);
80 
81  /* Decode size */
82  ret = hg_proc_uint64_t(proc, &buf_size);
83  if (ret != HG_SUCCESS) {
84  HG_LOG_ERROR("Proc error");
85  return ret;
86  }
87  if (!buf_size) {
88  /* If buf_size is 0, define handle to HG_BULK_NULL */
89  *bulk_ptr = HG_BULK_NULL;
90  break;
91  }
92 
93  buf = hg_proc_save_ptr(proc, buf_size);
94  ret = HG_Bulk_deserialize(hg_class, bulk_ptr, buf, buf_size);
95  if (ret != HG_SUCCESS) {
96  HG_LOG_ERROR("Could not deserialize bulk handle");
97  return ret;
98  }
99  /* Cache serialize ptr to buf */
100  ret = HG_Bulk_set_serialize_cached_ptr(*bulk_ptr, buf, buf_size);
101  if (ret != HG_SUCCESS) {
102  HG_LOG_ERROR("Could not set serialize cached ptr");
103  return ret;
104  }
105  hg_proc_restore_ptr(proc, buf, buf_size);
106  break;
107  }
108  case HG_FREE:
109  if (*bulk_ptr == HG_BULK_NULL) {
110  /* If *bulk is HG_BULK_NULL, just return success */
111  ret = HG_SUCCESS;
112  break;
113  }
114  /* Set serialize ptr to NULL */
115  ret = HG_Bulk_set_serialize_cached_ptr(*bulk_ptr, NULL, 0);
116  if (ret != HG_SUCCESS) {
117  HG_LOG_ERROR("Could not set serialize cached ptr");
118  return ret;
119  }
120  /* Decrement refcount on bulk handle */
121  ret = HG_Bulk_free(*bulk_ptr);
122  if (ret != HG_SUCCESS) {
123  HG_LOG_ERROR("Could not free bulk handle");
124  return ret;
125  }
126  *bulk_ptr = HG_BULK_NULL;
127  break;
128  default:
129  break;
130  }
131  return ret;
132 }
