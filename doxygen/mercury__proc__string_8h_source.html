144 static HG_INLINE hg_return_t
145 hg_proc_hg_string_object_t(hg_proc_t proc, void *string)
146 {
147  hg_uint64_t string_len = 0;
148  hg_return_t ret = HG_SUCCESS;
149  hg_string_object_t *strobj = (hg_string_object_t*)string;
150 
151  switch (hg_proc_get_op(proc)) {
152  case HG_ENCODE:
153  string_len = (strobj->data) ? strlen(strobj->data) + 1 : 0;
154  ret = hg_proc_uint64_t(proc, &string_len);
155  if (ret != HG_SUCCESS) {
156  HG_LOG_ERROR("Proc error");
157  goto done;
158  }
159  if (string_len) {
160  ret = hg_proc_raw(proc, strobj->data, string_len);
161  if (ret != HG_SUCCESS) {
162  HG_LOG_ERROR("Proc error");
163  goto done;
164  }
165  ret = hg_proc_hg_uint8_t(proc, (hg_uint8_t*) &strobj->is_const);
166  if (ret != HG_SUCCESS) {
167  HG_LOG_ERROR("Proc error");
168  goto done;
169  }
170  ret = hg_proc_hg_uint8_t(proc, (hg_uint8_t*) &strobj->is_owned);
171  if (ret != HG_SUCCESS) {
172  HG_LOG_ERROR("Proc error");
173  goto done;
174  }
175  }
176  break;
177  case HG_DECODE:
178  ret = hg_proc_uint64_t(proc, &string_len);
179  if (ret != HG_SUCCESS) {
180  HG_LOG_ERROR("Proc error");
181  goto done;
182  }
183  if (string_len) {
184  strobj->data = (char*) malloc(string_len);
185  ret = hg_proc_raw(proc, strobj->data, string_len);
186  if (ret != HG_SUCCESS) {
187  HG_LOG_ERROR("Proc error");
188  goto done;
189  }
190  ret = hg_proc_hg_uint8_t(proc, (hg_uint8_t*) &strobj->is_const);
191  if (ret != HG_SUCCESS) {
192  HG_LOG_ERROR("Proc error");
193  goto done;
194  }
195  ret = hg_proc_hg_uint8_t(proc, (hg_uint8_t*) &strobj->is_owned);
196  if (ret != HG_SUCCESS) {
197  HG_LOG_ERROR("Proc error");
198  goto done;
199  }
200  } else {
201  strobj->data = NULL;
202  }
203  break;
204  case HG_FREE:
205  ret = hg_string_object_free(strobj);
206  if (ret != HG_SUCCESS) {
207  HG_LOG_ERROR("Could not free string object");
208  goto done;
209  }
210  break;
211  default:
212  break;
213  }
214 
215 done:
216  return ret;
217 }
