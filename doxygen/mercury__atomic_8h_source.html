238 static HG_UTIL_INLINE hg_util_bool_t
239 hg_atomic_cas32(hg_atomic_int32_t *ptr, hg_util_int32_t compare_value,
240  hg_util_int32_t swap_value)
241 {
242  hg_util_bool_t ret;
243 
244 #if defined(_WIN32)
245  ret = (compare_value == InterlockedCompareExchangeNoFence(&ptr->value,
246  swap_value, compare_value));
247 #elif defined(HG_UTIL_HAS_OPA_PRIMITIVES_H)
248  ret = (hg_util_bool_t) (compare_value == OPA_cas_int(ptr, compare_value, swap_value));
249 #elif defined(HG_UTIL_HAS_STDATOMIC_H)
250  ret = atomic_compare_exchange_strong_explicit(ptr, &compare_value,
251  swap_value, memory_order_acq_rel, memory_order_acquire);
252 #elif defined(__APPLE__)
253  ret = OSAtomicCompareAndSwap32(compare_value, swap_value, &ptr->value);
254 #else
255  #error "Not supported on this platform."
256 #endif
257 
258  return ret;
259 }
