161 /*---------------------------------------------------------------------------*/
162 static HG_UTIL_INLINE int
163 hg_atomic_queue_push(struct hg_atomic_queue *hg_atomic_queue, void *entry)
164 {
165  hg_util_int32_t prod_head, prod_next, cons_tail;
166 
167  do {
168  prod_head = hg_atomic_get32(&hg_atomic_queue->prod_head);
169  prod_next = (prod_head + 1) & (int) hg_atomic_queue->prod_mask;
170  cons_tail = hg_atomic_get32(&hg_atomic_queue->cons_tail);
171 
172  if (prod_next == cons_tail) {
173  hg_atomic_fence();
174  if (prod_head == hg_atomic_get32(&hg_atomic_queue->prod_head) &&
175  cons_tail == hg_atomic_get32(&hg_atomic_queue->cons_tail)) {
176  hg_atomic_queue->drops++;
177  /* Full */
178  return HG_UTIL_FAIL;
179  }
180  continue;
181  }
182  } while (
183  !hg_atomic_cas32(&hg_atomic_queue->prod_head, prod_head, prod_next));
184 
185  hg_atomic_set64(&hg_atomic_queue->ring[prod_head], (hg_util_int64_t) entry);
186 
187  /*
188  * If there are other enqueues in progress
189  * that preceded us, we need to wait for them
190  * to complete
191  */
192  while (hg_atomic_get32(&hg_atomic_queue->prod_tail) != prod_head)
193  cpu_spinwait();
194 
195  hg_atomic_set32(&hg_atomic_queue->prod_tail, prod_next);
196 
197  return HG_UTIL_SUCCESS;
198 }

201 static HG_UTIL_INLINE void *
202 hg_atomic_queue_pop_mc(struct hg_atomic_queue *hg_atomic_queue)
203 {
204  hg_util_int32_t cons_head, cons_next;
205  void *entry = NULL;
206 
207  do {
208  cons_head = hg_atomic_get32(&hg_atomic_queue->cons_head);
209  cons_next = (cons_head + 1) & (int) hg_atomic_queue->cons_mask;
210 
211  if (cons_head == hg_atomic_get32(&hg_atomic_queue->prod_tail))
212  return NULL;
213  } while (
214  !hg_atomic_cas32(&hg_atomic_queue->cons_head, cons_head, cons_next));
215 
216  entry = (void *) hg_atomic_get64(&hg_atomic_queue->ring[cons_head]);
217 
218  /*
219  * If there are other dequeues in progress
220  * that preceded us, we need to wait for them
221  * to complete
222  */
223  while (hg_atomic_get32(&hg_atomic_queue->cons_tail) != cons_head)
224  cpu_spinwait();
225 
226  hg_atomic_set32(&hg_atomic_queue->cons_tail, cons_next);
227 
228  return entry;
229 }

232 static HG_UTIL_INLINE void *
233 hg_atomic_queue_pop_sc(struct hg_atomic_queue *hg_atomic_queue)
234 {
235  hg_util_int32_t cons_head, cons_next;
236  hg_util_int32_t prod_tail;
237  void *entry = NULL;
238 
239  cons_head = hg_atomic_get32(&hg_atomic_queue->cons_head);
240  prod_tail = hg_atomic_get32(&hg_atomic_queue->prod_tail);
241  cons_next = (cons_head + 1) & (int) hg_atomic_queue->cons_mask;
242 
243  if (cons_head == prod_tail)
244  /* Empty */
245  return NULL;
246 
247  hg_atomic_set32(&hg_atomic_queue->cons_head, cons_next);
248 
249  entry = (void *) hg_atomic_get64(&hg_atomic_queue->ring[cons_head]);
250 
251  hg_atomic_set32(&hg_atomic_queue->cons_tail, cons_next);
252 
253  return entry;
254 }
