156 /*---------------------------------------------------------------------------*/
157 static HG_UTIL_INLINE int
158 hg_atomic_queue_push(struct hg_atomic_queue *hg_atomic_queue, void *entry)
159 {
160  hg_util_int32_t prod_head, prod_next, cons_tail;
161  int ret = HG_UTIL_SUCCESS;
162 
163  do {
164  prod_head = hg_atomic_get32(&hg_atomic_queue->prod_head);
165  prod_next = (prod_head + 1) & (int) hg_atomic_queue->prod_mask;
166  cons_tail = hg_atomic_get32(&hg_atomic_queue->cons_tail);
167 
168  if (prod_next == cons_tail) {
169  hg_atomic_fence();
170  if (prod_head == hg_atomic_get32(&hg_atomic_queue->prod_head) &&
171  cons_tail == hg_atomic_get32(&hg_atomic_queue->cons_tail)) {
172  hg_atomic_queue->drops++;
173  /* Full */
174  ret = HG_UTIL_FAIL;
175  goto done;
176  }
177  continue;
178  }
179  } while (!hg_atomic_cas32(&hg_atomic_queue->prod_head, prod_head,
180  prod_next));
181 
182  hg_atomic_set64((hg_atomic_int64_t *) &hg_atomic_queue->ring[prod_head],
183  (hg_util_int64_t) entry);
184 
185  /*
186  * If there are other enqueues in progress
187  * that preceded us, we need to wait for them
188  * to complete
189  */
190  while (hg_atomic_get32(&hg_atomic_queue->prod_tail) != prod_head)
191  cpu_spinwait();
192 
193  hg_atomic_set32(&hg_atomic_queue->prod_tail, prod_next);
194 
195 done:
196  return ret;
197 }

199 /*---------------------------------------------------------------------------*/
200 static HG_UTIL_INLINE void *
201 hg_atomic_queue_pop_mc(struct hg_atomic_queue *hg_atomic_queue)
202 {
203  hg_util_int32_t cons_head, cons_next;
204  void *entry = NULL;
205 
206  do {
207  cons_head = hg_atomic_get32(&hg_atomic_queue->cons_head);
208  cons_next = (cons_head + 1) & (int) hg_atomic_queue->cons_mask;
209 
210  if (cons_head == hg_atomic_get32(&hg_atomic_queue->prod_tail))
211  goto done;
212  } while (!hg_atomic_cas32(&hg_atomic_queue->cons_head, cons_head,
213  cons_next));
214 
215  entry = (void *) hg_atomic_get64(
216  (hg_atomic_int64_t *) &hg_atomic_queue->ring[cons_head]);
217 
218  /*
219  * If there are other dequeues in progress
220  * that preceded us, we need to wait for them
221  * to complete
222  */
223  while (hg_atomic_get32(&hg_atomic_queue->cons_tail) != cons_head)
224  cpu_spinwait();
225 
226  hg_atomic_set32(&hg_atomic_queue->cons_tail, cons_next);
227 
228 done:
229  return entry;
230 }

232 /*---------------------------------------------------------------------------*/
233 static HG_UTIL_INLINE void *
234 hg_atomic_queue_pop_sc(struct hg_atomic_queue *hg_atomic_queue)
235 {
236  hg_util_int32_t cons_head, cons_next;
237  hg_util_int32_t prod_tail;
238  void *entry = NULL;
239 
240  cons_head = hg_atomic_get32(&hg_atomic_queue->cons_head);
241  prod_tail = hg_atomic_get32(&hg_atomic_queue->prod_tail);
242  cons_next = (cons_head + 1) & (int) hg_atomic_queue->cons_mask;
243 
244  if (cons_head == prod_tail)
245  /* Empty */
246  goto done;
247 
248  hg_atomic_set32(&hg_atomic_queue->cons_head, cons_next);
249 
250  entry = (void *) hg_atomic_get64(
251  (hg_atomic_int64_t *) &hg_atomic_queue->ring[cons_head]);
252 
253  hg_atomic_set32(&hg_atomic_queue->cons_tail, cons_next);
254 
255 done:
256  return entry;
257 }
